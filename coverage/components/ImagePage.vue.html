<template>
  <div class="container">
    <h1 class="content__header">{{section.name}}</h1>
    <div id="thumb-container" class="thumb-container">
      <div class="drawbox image__wrap"
           v-for="image in section.images"
           :key="image.url"
           @click="bigimageid = `${ image.url }`,
                    sectionType = `${section.type}`,
                    sectionSlug = `${section.slug}`,
                    bigImage($event)">
        <v-lazy-image
          :src="require(`~/assets/images/${section.type}/${ section.slug }/${ image.url }@0.25x.jpg`)"
          :src-placeholder="require(`~/assets/images/${section.type}/${ section.slug }/${ image.url }@lqp.png`)"
          :alt="image.alt"
          :id="image.url"
          class = "draw__image image__thumb"
        ></v-lazy-image>
        <div class="draw image__text">
          <h3 class="image__header draw__header"
              :class="[section.slug]">{{ image.title }}</h3>
        </div>
      </div>
    </div>
  </div>
</template>
 
<script>
  import { gsap } from 'gsap'
  import photography from '@/data/photography'
  export default {
    data() {
      return {
        clicked: false,
        bigimageid: 'blank',
        isActive: true,
        sectionSlug: 'blank',
        sectionType: 'blank'
      }
    },
    computed: {
      section() {
        return photography.sections.find(
          section => section.slug === this.$route.params.id
        );
      },
      windowWidth() {
        let windowWidth;
        return windowWidth = window.innerWidth;
      }
    },
    methods: {
      bigImage($event) {
        if(this.windowWidth > 768){
          console.log('event is ', $event);
          let whichImage = this.bigimageid;
          let tl = gsap.timeline();
          tl.to(`.draw__image`, {duration: 0.1, display: 'inline-block', height: 200});
          tl.to(`#${ whichImage }`, {duration: 0.2, height: '70vh'});
          tl.to(window, {duration: 0.2, scrollTo: {y: `#${ whichImage }`, offsetY: 110}, delay: 0.23});
          let thumbs = document.querySelectorAll('.image__wrap');
          for (let i = 0; i < thumbs.length; i++) { thumbs[i].classList.remove('image__big'); }
          let theThing = document.querySelector(`#${ whichImage }`);
          const slug = `https://rabbits.s3.eu-west-2.amazonaws.com/${ this.sectionType }/${ this.sectionSlug }/${ this.bigimageid }.jpg`
          theThing.src = slug;
          let parentThing = theThing.closest('.image__wrap');
          parentThing.classList.add('image__big');
        }
      }
    }
  }
</script>
